CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
SET(KLAMPT_ROOT /home/aorthey/git/Klampt/)
SET (CMAKE_MODULE_PATH "${KLAMPT_ROOT}/CMakeModules")
FIND_PACKAGE(Klampt REQUIRED) 
ADD_DEFINITIONS(${KLAMPT_DEFINITIONS})
INCLUDE_DIRECTORIES(${KLAMPT_INCLUDE_DIRS})

SET(EXECUTABLES 
        test/display_environment.cpp 
        main.cpp
)

MESSAGE("-------------------------------------")
FOREACH(FILENAME ${EXECUTABLES})
        GET_FILENAME_COMPONENT( EXECUTABLE ${FILENAME} NAME_WE)
        MESSAGE("${FILENAME} >> ${EXECUTABLE}")
        ADD_EXECUTABLE( ${EXECUTABLE} ${FILENAME})
        TARGET_LINK_LIBRARIES(${EXECUTABLE} ${KLAMPT_LIBRARIES})
ENDFOREACH(FILENAME)
MESSAGE("-------------------------------------")
